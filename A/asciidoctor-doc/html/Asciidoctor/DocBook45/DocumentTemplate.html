def title_tags(str)
  if str.include?(': ')
    title, _, subtitle = str.rpartition(': ')
    %(<title>#{title}</title>
  <subtitle>#{subtitle}</subtitle>)
  else
    %(<title>#{str}</title>)
  end
end
